114  protected String encodeInternal(byte[] data) {
115  if (data.length == 0)
116  return "";
117 
118  // SHIFT is the number of bits per output character, so the length of
119  // the
120  // output is the length of the input multiplied by 8/SHIFT, rounded up.
121  if (data.length >= (1 << 28))
122  // The computation below will fail, so don't do it.
123  throw new IllegalArgumentException();
124 
125  final int outputLength = (data.length * 8 + SHIFT - 1) / SHIFT;
126  final StringBuilder result = new StringBuilder(outputLength);
127 
128  int buffer = data[0];
129  int next = 1;
130  int bitsLeft = 8;
131  while (bitsLeft > 0 || next < data.length) {
132  if (bitsLeft < SHIFT)
133  if (next < data.length) {
134  buffer <<= 8;
135  buffer |= (data[next++] & 0xff);
136  bitsLeft += 8;
137  } else {
138  final int pad = SHIFT - bitsLeft;
139  buffer <<= pad;
140  bitsLeft += pad;
141  }
142  final int index = MASK & (buffer >> (bitsLeft - SHIFT));
143  bitsLeft -= SHIFT;
144  result.append(DIGITS[index]);
145  }
146  return result.toString();
147  }
