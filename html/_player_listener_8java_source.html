61  public boolean isFrozen(Player player) {
62  if (extraauth.DB.IsAuth(player))
63  return false;
64  if (!(Boolean) extraauth.Settings._("FreezePlayer"))
65  return false;
66 
67  if (messageTimeout.containsKey(player.getUniqueId())) {
68  final long time = System.currentTimeMillis()
69  - messageTimeout.get(player.getUniqueId()).longValue();
70  if (time >= 2000) {
71  player.sendMessage(ChatColor.YELLOW + "[ExtraAuth] " + ChatColor.GOLD
72  + extraauth.Lang._("FreezeMessage"));
73  messageTimeout.put(player.getUniqueId(), time);
74  }
75  } else {
76  player.sendMessage(ChatColor.YELLOW + "[ExtraAuth] " + ChatColor.GOLD
77  + extraauth.Lang._("FreezeMessage"));
78  messageTimeout.put(player.getUniqueId(), System.currentTimeMillis());
79  }
80 
81  return true;
82  }
