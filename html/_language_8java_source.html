35  public Language(ExtraAuth extraauth) {

38  + File.separator + "lang" + File.separator;
39 
40  try {
41  path = new File(partpath + extraauth.Settings._("Language") + ".yml");
42  if (path.exists())
43  file.load(path);
44  else {
45  extraauth._W("Couldn't find language file, reverting to en-US", false);

54  }
55  }
