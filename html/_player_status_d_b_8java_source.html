125  public FailedReason Auth(Player player, String key) {
126  final playerstatus p = Get(player.getName());
127  if (p == null)
128  return FailedReason.NOT_REGISTERED;
129 
130  if (p.Authed)
131  return FailedReason.ALREADY_AUTHED;
132 
133  try {
134  switch (p.Method) {
135  case TOTP:
136  final long time = System.currentTimeMillis() / 30000L;
137  final String PrivateKey = p.PrivateKey;
138  if (TOTP.GenerateTOTP(PrivateKey, time, 6, 1).trim()
139  .equalsIgnoreCase(key.trim())) {
140  p.Authed = true;
141  player.setFlying(false);
142  return FailedReason.SUCCESSFULL;
143  } else
144  return FailedReason.WRONG_KEY;
145  case KEY:
146  if (key.equals(p.PrivateKey)) {
147  p.Authed = true;
148  return FailedReason.SUCCESSFULL;
149  } else
150  return FailedReason.WRONG_KEY;
151  default:
152  return FailedReason.INVALID_METHOD;
153  }
154 
155  } catch (final Exception e) {
156  e.printStackTrace();
157  }
158  return FailedReason.INVALID_METHOD;
159  }

213  public void Load() {
214  try {

216  file));

218  if (version != CURRENT_VERSION)
